<!DOCTYPE html>
<html>
<head>
    <title>Mapa de Nueva York con Capas</title>
    <meta charset="utf-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.7.1/dist/leaflet.css" />
    <script src="https://unpkg.com/leaflet@1.7.1/dist/leaflet.js"></script>
</head>
<body>
    <div id="map" style="width: 100%; height: 600px;"></div>
    <script>
        // Inicializar el mapa centrado en Nueva York
        var map = L.map('map').setView([40.7128, -74.0060], 12);

        // Añadir capa base de OpenStreetMap
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
        }).addTo(map);

        // Datos GeoJSON de estaciones de metro
        var subwayData = {
            "type": "FeatureCollection",
            "features": [
                {
                    "type": "Feature",
                    "geometry": {
                        "type": "Point",
                        "coordinates": [-74.005941, 40.712784]
                    },
                    "properties": {
                        "name": "Estación 1"
                    }
                },
                {
                    "type": "Feature",
                    "geometry": {
                        "type": "Point",
                        "coordinates": [-73.985428, 40.748817]
                    },
                    "properties": {
                        "name": "Estación 2"
                    }
                }
                // Más datos...
            ]
        };

        // Datos GeoJSON de parques
        var parksData = {
            "type": "FeatureCollection",
            "features": [
                {
                    "type": "Feature",
                    "geometry": {
                        "type": "Polygon",
                        "coordinates": [[
                            [-74.012084, 40.705569],
                            [-74.013642, 40.706225],
                            [-74.013281, 40.707119],
                            [-74.011751, 40.706493],
                            [-74.012084, 40.705569]
                        ]]
                    },
                    "properties": {
                        "name": "Parque 1"
                    }
                },
                {
                    "type": "Feature",
                    "geometry": {
                        "type": "Polygon",
                        "coordinates": [[
                            [-73.968285, 40.785091],
                            [-73.958285, 40.785091],
                            [-73.958285, 40.795091],
                            [-73.968285, 40.795091],
                            [-73.968285, 40.785091]
                        ]]
                    },
                    "properties": {
                        "name": "Parque 2"
                    }
                }
                // Más datos...
            ]
        };

        // Estilo para las estaciones de metro
        function subwayStyle(feature) {
            return {
                radius: 8,
                fillColor: "#0000FF", // Azul
                color: "#000",
                weight: 1,
                opacity: 1,
                fillOpacity: 0.8
            };
        }

        // Estilo para los parques
        function parksStyle(feature) {
            return {
                color: "#00FF00", // Verde
                weight: 2,
                opacity: 1,
                fillOpacity: 0.2
            };
        }

        // Añadir capa de estaciones de metro
        var subwayLayer = L.geoJson(subwayData, {
            pointToLayer: function (feature, latlng) {
                return L.circleMarker(latlng, subwayStyle(feature));
            },
            onEachFeature: function (feature, layer) {
                if (feature.properties && feature.properties.name) {
                    layer.bindPopup(feature.properties.name);
                }
            }
        });
        subwayLayer.addTo(map);

        // Añadir capa de parques
        var parksLayer = L.geoJson(parksData, {
            style: parksStyle,
            onEachFeature: function (feature, layer) {
                if (feature.properties && feature.properties.name) {
                    layer.bindPopup(feature.properties.name);
                }
            }
        });
        parksLayer.addTo(map);

        // Añadir control de capas
        var layerControl = L.control.layers(null, {
            'Estaciones de Metro': subwayLayer,
            'Parques': parksLayer
        }, {
            collapsed: false
        }).addTo(map);
    </script>
</body>
</html>
